CREATE TABLE `user` (
`id` INTEGER PRIMARY KEY AUTOINCREMENT
);

CREATE TABLE `links` (
`id` INTEGER PRIMARY KEY AUTOINCREMENT, 
user_id INTEGER,
comment INTEGER,
FOREIGN KEY(user_id) REFERENCES user(id)
FOREIGN KEY(comment_id) REFERENCES comment(id)
);

CREATE TABLE `comments` (
`id` INTEGER PRIMARY KEY AUTOINCREMENT, 
user_id INTEGER,
FOREIGN KEY(user_id) REFERENCES user(id)
);
